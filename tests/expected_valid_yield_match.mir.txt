functions:
  - public fn parse -> Config yields ParseFailure
      locals:
        %0[raw] : Text
        %1[$return] : Config
        %2[reason] : ParseFailure
      block %0
        %2[reason] = construct ParseFailure::Bad {}
        fail %2[reason]
  - public fn boot -> Config
      locals:
        %0[raw] : Text
        %1[$return] : Config
        %2[success] : Config
        %3[failure] : ParseFailure
        %4[config] : Config
        %5[key] : Text
      block %0
        invoke parse(%0[raw]) ok %2[success] -> %1 fail %3[failure] -> %2
      block %1
        %4[config] = %2[success]
        %1[$return] = %4[config]
        goto %3
      block %2
        switch %3[failure]
          arm Bad -> %4
          arm MissingKey -> %5
      block %3
        return %1[$return]
      block %4
        %1[$return] = construct Config {raw: "fallback"}
        goto %3
      block %5
        %5[key] = %3[failure].key
        %1[$return] = construct Config {raw: %5[key]}
        goto %3
